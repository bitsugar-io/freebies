# Duplicate Notifications & Retry Logic Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development
> (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use
> checkbox (`- [ ]`) syntax for tracking.

**Goal:** Fix duplicate push notifications caused by multiple user records sharing the same Expo push
token, and add retry with exponential backoff to all external HTTP calls (Expo push API, MLB stats
API).

**Architecture:** Shared `httputil` package provides retry-with-backoff for HTTP calls, used by both
the Expo notifier and MLB source client. Push token ownership is enforced at the DB level (partial
unique index) and application level (clear stale tokens on registration). Message deduplication by
push token prevents duplicates at send time. Mobile app uses stable device identifiers.

**Tech Stack:** Go (API server), SQLite/Turso (DB), sqlc (query codegen), goose (migrations),
React Native / Expo (mobile), testify (Go tests)

**Spec:** `docs/superpowers/specs/2026-03-18-duplicate-notifications-and-retry-design.md`

```go
package httputil

import (
	"context"
	"fmt"
	"math"
	"math/rand/v2"
	"net/http"
	"time"
)

// RetryOptions configures retry behavior for HTTP requests.
type RetryOptions struct {
	MaxRetries int           // Maximum number of retries (default 3)
	BaseDelay  time.Duration // Base delay for exponential backoff (default 1s)
}

func (o *RetryOptions) maxRetries() int {
	if o != nil && o.MaxRetries > 0 {
		return o.MaxRetries
	}
	return 3
}

func (o *RetryOptions) baseDelay() time.Duration {
	if o != nil && o.BaseDelay > 0 {
		return o.BaseDelay
	}
	return time.Second
}

// Do executes an HTTP request with retry logic and exponential backoff.
// The newReq function is called for each attempt to produce a fresh *http.Request,
// avoiding consumed-body issues on retry.
// Retries on network errors, 5xx, and 429 (rate limited). Does not retry other 4xx.
func Do(ctx context.Context, client *http.Client, newReq func() (*http.Request, error), opts *RetryOptions) (*http.Response, error) {
	maxRetries := opts.maxRetries()
	baseDelay := opts.baseDelay()

	var lastErr error

	for attempt := 0; attempt <= maxRetries; attempt++ {
		if attempt > 0 {
			delay := backoffWithJitter(baseDelay, attempt)
			select {
			case <-ctx.Done():
				return nil, fmt.Errorf("context cancelled during retry backoff: %w", ctx.Err())
			case <-time.After(delay):
			}
		}

		req, err := newReq()
		if err != nil {
			return nil, fmt.Errorf("building request: %w", err)
		}

		resp, err := client.Do(req)
		if err != nil {
			lastErr = err
			continue // network error, retry
		}

		if !shouldRetry(resp.StatusCode) || attempt == maxRetries {
			return resp, nil // return as-is (caller must close body)
		}

		// Close body before retry to avoid leaking connections
		resp.Body.Close()
		lastErr = nil
	}

	// Only reached on network errors exhausting all retries
	return nil, lastErr
}

func shouldRetry(statusCode int) bool {
	if statusCode == http.StatusTooManyRequests {
		return true
	}
	return statusCode >= 500
}

func backoffWithJitter(base time.Duration, attempt int) time.Duration {
	delay := base * time.Duration(math.Pow(2, float64(attempt-1)))
	// Add +/-25% jitter
	jitter := 0.75 + rand.Float64()*0.5
	return time.Duration(float64(delay) * jitter)
}
```

func (n *ExpoNotifier) SendBatch(ctx context.Context, messages []ExpoPushMessage) ([]ExpoPushTicket, error) {
	if len(messages) == 0 {
		return nil, nil
	}

	payload, err := json.Marshal(messages)
	if err != nil {
		return nil, fmt.Errorf("marshaling messages: %w", err)
	}

	newReq := func() (*http.Request, error) {
		req, err := http.NewRequestWithContext(ctx, "POST", n.url, bytes.NewReader(payload))
		if err != nil {
			return nil, err
		}
		req.Header.Set("Content-Type", "application/json")
		req.Header.Set("Accept", "application/json")
		return req, nil
	}

	resp, err := httputil.Do(ctx, n.client, newReq, nil)
	if err != nil {
		return nil, fmt.Errorf("sending request: %w", err)
	}
	defer resp.Body.Close()

	if resp.StatusCode != http.StatusOK {
		return nil, fmt.Errorf("unexpected status: %d", resp.StatusCode)
	}

	var pushResp ExpoPushResponse
	if err := json.NewDecoder(resp.Body).Decode(&pushResp); err != nil {
		return nil, fmt.Errorf("decoding response: %w", err)
	}

	return pushResp.Data, nil
}

// DeduplicateMessages removes duplicate messages by push token, keeping the first occurrence.
func DeduplicateMessages(messages []ExpoPushMessage) []ExpoPushMessage {
	seen := make(map[string]bool, len(messages))
	result := make([]ExpoPushMessage, 0, len(messages))
	for _, msg := range messages {
		if !seen[msg.To] {
			seen[msg.To] = true
			result = append(result, msg)
		}
	}
	return result
}

-- +goose Up
-- Fix duplicate push tokens: keep newest user per token, transfer subscriptions, clean up

-- Step 1: Transfer subscriptions from old duplicate users to the newest user per push_token.
-- For each duplicate push_token group, the newest user (by created_at) is kept.
INSERT OR IGNORE INTO subscriptions (id, user_id, event_id)
SELECT
    'migrated-' || s.id,
    keeper.id,
    s.event_id
FROM subscriptions s
JOIN users u ON s.user_id = u.id
JOIN (
    SELECT push_token, MAX(created_at) as max_created
    FROM users
    WHERE push_token IS NOT NULL AND push_token != ''
    GROUP BY push_token
    HAVING COUNT(*) > 1
) dupes ON u.push_token = dupes.push_token
JOIN users keeper ON keeper.push_token = dupes.push_token AND keeper.created_at = dupes.max_created
WHERE u.id != keeper.id;

-- Step 2: NULL out push_token on older duplicate users
UPDATE users SET push_token = NULL, updated_at = CURRENT_TIMESTAMP
WHERE id IN (
    SELECT u.id
    FROM users u
    JOIN (
        SELECT push_token, MAX(created_at) as max_created
        FROM users
        WHERE push_token IS NOT NULL AND push_token != ''
        GROUP BY push_token
        HAVING COUNT(*) > 1
    ) dupes ON u.push_token = dupes.push_token
    WHERE u.created_at < dupes.max_created
);

CREATE UNIQUE INDEX IF NOT EXISTS idx_users_push_token_unique
